import React, { useState, useMemo, useEffect } from 'react';
import { 
  Search, Folder, Plus, X, Link, Bookmark, ExternalLink, 
  AlignLeft, Bell, User, Settings, Filter, Grid, List, MoreVertical,
  Copy, Edit3, Trash2, Bot, Loader2, PenLine
} from 'lucide-react';

  { id: 'all', name: '전체 지식', icon: Folder },
  { id: 'dev', name: '개발', icon: Folder },
  { id: 'design', name: '디자인', icon: Folder },
  { id: 'ai', name: 'AI & Data', icon: Folder },
  { id: 'idea', name: '아이디어 보관함', icon: Folder }
];

  {
    id: 1,

    details: 'React의 렌더링 과정은 Render 단계와 Commit 단계로 나뉜다. useEffect 훅을 사용할 때 의존성 배열(dependency array)을 관리하는 것이 성능 최적화의 핵심이다.',
    memo: '내일 팀 미팅 때 useEffect 의존성 배열 관리 기법 파트 공유할 것.'
  },

    title: 'GPT-4 기술 리포트 및 활용 사례',

    summary: 'GPT-4의 다중 모달(Multi-modal) 기능과 향상된 추론 능력에 대한 요약 보고서.',
    details: 'GPT-4는 기존 텍스트뿐만 아니라 이미지를 입력으로 받을 수 있으며, 변호사 시험에서 상위 10%의 성적을 기록할 만큼 향상된 추론 능력을 보여준다. 프롬프트 엔지니어링의 중요성이 더욱 커졌다.',

    summary: '글래스모피즘의 진화, 마이크로 인터랙션, 다크 모드 최적화에 대한 디자인 레퍼런스 모음.',
    details: '사용자의 시선을 사로잡기 위해 미세한 애니메이션(마이크로 인터랙션)의 사용이 증가하고 있으며, 공간감을 주는 은은한 그림자와 반투명한 패널 디자인이 주류를 이루고 있다.',
    memo: '다음 사이드 프로젝트 UI 기획할 때 글래스모피즘 적용해보기.'
  }
];

export default function App() {
  // ==========================================
  // [AI-SECTION: STATE MANAGEMENT]
  // ==========================================
  const [cards, setCards] = useState(INITIAL_CARDS);
  const [activeFolder, setActiveFolder] = useState('all');
  const [searchQuery, setSearchQuery] = useState('');
  const [urlInput, setUrlInput] = useState('');
  const [selectedCard, setSelectedCard] = useState(null);
  
  const [isAnalyzing, setIsAnalyzing] = useState(false);

  // ==========================================
  // [AI-SECTION: UPDATE LOGIC (메모 수정용)]
  // 특정 카드의 데이터를 업데이트하는 공통 함수
  // ==========================================
  const handleUpdateCard = (cardId, updatedFields) => {
    setCards(prevCards => 
      prevCards.map(card => card.id === cardId ? { ...card, ...updatedFields } : card)
    );
    // 현재 열려있는 모달창의 데이터도 동기화
    if (selectedCard?.id === cardId) {
      setSelectedCard(prev => ({ ...prev, ...updatedFields }));
    }
  };

  // ==========================================
  // [AI-SECTION: ASYNC SIMULATION LOGIC]
  // ==========================================
  const handleAnalyzeUrl = () => {
    if (!urlInput.trim()) return;
    
    setIsAnalyzing(true);
    const submittedUrl = urlInput;
    setUrlInput('');

    setTimeout(() => {
      const newCard = {
        id: Date.now(),
        folderId: activeFolder === 'all' ? 'idea' : activeFolder,
        url: submittedUrl,
        title: '✨ AI가 자동 분석한 새로운 문서',
        tags: ['AI 추출', '신규 자료'],
        summary: '입력하신 URL을 바탕으로 AI가 백그라운드에서 비동기적으로 추출 및 요약한 내용입니다.',
        details: '이곳에는 크롤링된 전체 텍스트를 바탕으로 LLM(GPT 등)이 작성한 상세한 마크다운 형태의 분석 글이 들어갑니다. SQLAlchemy를 통해 RDBMS에 저장될 핵심 데이터입니다.',
        memo: '' // 초기 메모는 빈 값으로 생성
      };

      setCards(prevCards => [newCard, ...prevCards]);
      setIsAnalyzing(false);
    }, 3000);
  };

  const handleKeyDown = (e) => {
    if (e.key === 'Enter') {
      handleAnalyzeUrl();
    }
  };

  // ==========================================
  // [AI-SECTION: LOGIC & FILTERING]
  // ==========================================
  const filteredCards = useMemo(() => {
    return cards.filter(card => {
      const matchFolder = activeFolder === 'all' || card.folderId === activeFolder;
      const query = searchQuery.toLowerCase();
      const matchSearch = 
        card.title.toLowerCase().includes(query) || 
        card.summary.toLowerCase().includes(query) ||
        card.tags.some(tag => tag.toLowerCase().includes(query));

      return matchFolder && matchSearch;
    });
  }, [cards, activeFolder, searchQuery]);

  const handleCopyLink = (url, e) => {
    e.stopPropagation();
    navigator.clipboard.writeText(url);
  };

  // ==========================================
  // [AI-SECTION: COMPONENTS]
  // ==========================================

  const SkeletonCard = () => (
    <div className="bg-white rounded-2xl border border-gray-200 p-6 flex flex-col h-[280px] animate-pulse shadow-sm">
      <div className="flex justify-between items-start mb-4">
        <div className="flex gap-2">
          <div className="w-12 h-6 bg-blue-50 rounded-md"></div>
          <div className="w-16 h-6 bg-blue-50 rounded-md"></div>
        </div>
        <div className="w-8 h-8 bg-gray-100 rounded-full flex items-center justify-center">
          <Loader2 className="w-4 h-4 text-blue-400 animate-spin" />
        </div>
      </div>
      <div className="w-3/4 h-6 bg-gray-200 rounded mb-3"></div>
      <div className="w-full h-4 bg-gray-100 rounded mb-2"></div>
      <div className="w-full h-4 bg-gray-100 rounded mb-2"></div>
      <div className="w-2/3 h-4 bg-gray-100 rounded flex-1"></div>
      <div className="mt-auto pt-4 border-t border-gray-100">
        <div className="w-1/2 h-4 bg-gray-200 rounded"></div>
      </div>
    </div>
  );

  // 2. 디테일 모달창 (메모장 영역 추가됨)
  const DetailModal = ({ card, onClose, onUpdateCard }) => {
    // 내부 메모 편집 상태 관리
    const [memoText, setMemoText] = useState(card?.memo || '');
    const [isEditingMemo, setIsEditingMemo] = useState(false);

    // 카드가 변경되면 메모장 텍스트 동기화
    useEffect(() => {
      if (card) {
        setMemoText(card.memo || '');
        setIsEditingMemo(false);
      }
    }, [card]);

    if (!card) return null;

    const handleSaveMemo = () => {
      onUpdateCard(card.id, { memo: memoText });
      setIsEditingMemo(false);
    };

    return (
      <div className="fixed inset-0 bg-gray-900/60 backdrop-blur-sm flex items-center justify-center z-50 p-4 sm:p-6">
        <div className="bg-white rounded-2xl w-full max-w-3xl max-h-[90vh] overflow-hidden shadow-2xl flex flex-col animate-in fade-in zoom-in-95 duration-200">
          
          <div className="flex-none bg-white border-b border-gray-100 px-6 py-4 flex justify-between items-center z-10">
            <div className="flex items-center gap-3">
              <div className="w-10 h-10 rounded-full bg-blue-50 flex items-center justify-center shrink-0">
                <Bookmark className="w-5 h-5 text-blue-600" />
              </div>
              <h2 className="text-xl font-bold text-gray-800 line-clamp-1">{card.title}</h2>
            </div>
            <div className="flex items-center gap-2 shrink-0 ml-4">
              <button className="p-2 text-gray-400 hover:text-gray-600 hover:bg-gray-100 rounded-lg transition-colors" title="수정">
                <Edit3 className="w-5 h-5" />
              </button>
              <button className="p-2 text-gray-400 hover:text-red-600 hover:bg-red-50 rounded-lg transition-colors" title="삭제">
                <Trash2 className="w-5 h-5" />
              </button>
              <div className="w-px h-6 bg-gray-200 mx-1"></div>
              <button onClick={onClose} className="p-2 hover:bg-gray-100 rounded-full transition-colors">
                <X className="w-6 h-6 text-gray-500" />
              </button>
            </div>
          </div>

          <div className="flex-1 overflow-y-auto p-6 md:p-8">
            <div className="max-w-2xl mx-auto space-y-8">
              <div>
                <div className="flex flex-wrap items-center gap-3 mb-4">
                  <span className="flex items-center text-sm font-medium text-blue-700 bg-blue-50 px-3 py-1.5 rounded-lg border border-blue-100">
                    <Link className="w-4 h-4 mr-2" />
                    <a href={card.url} target="_blank" rel="noreferrer" className="hover:underline break-all">{card.url}</a>
                  </span>
                  <button onClick={(e) => handleCopyLink(card.url, e)} className="text-gray-400 hover:text-gray-700 text-sm flex items-center gap-1">
                    <Copy className="w-4 h-4" /> 복사
                  </button>
                </div>
                <div className="flex flex-wrap gap-2">
                  {card.tags.map(tag => (
                    <span key={tag} className="px-3 py-1 bg-gray-100 text-gray-700 text-sm rounded-md font-medium">
                      #{tag}
                    </span>
                  ))}
                </div>
              </div>

              <section>
                <h3 className="text-sm font-bold text-gray-400 mb-3 flex items-center gap-2">
                  <AlignLeft className="w-4 h-4" /> AI 요약 (Summary)
                </h3>
                <p className="text-gray-800 text-lg leading-relaxed bg-gradient-to-br from-gray-50 to-white p-5 rounded-xl border border-gray-200 shadow-sm">
                  {card.summary}
                </p>
              </section>

              <section>
                <h3 className="text-sm font-bold text-gray-400 mb-3 flex items-center gap-2">
                  <AlignLeft className="w-4 h-4" /> 세부 내용 (Details)
                </h3>
                <div className="text-gray-700 leading-relaxed whitespace-pre-wrap bg-white p-1 rounded-xl">
                  {card.details}
                </div>
              </section>

              {/* [AI-SECTION: 추가된 사용자 메모 영역] */}
              <section className="pt-4 border-t border-gray-100">
                <div className="flex items-center justify-between mb-3">
                  <h3 className="text-sm font-bold text-gray-400 flex items-center gap-2">
                    <PenLine className="w-4 h-4" /> 사용자 메모 (Memo)
                  </h3>
                  {!isEditingMemo && (
                    <button 
                      onClick={() => setIsEditingMemo(true)} 
                      className="text-xs text-blue-600 hover:text-blue-800 font-medium px-2 py-1 bg-blue-50 rounded hover:bg-blue-100 transition-colors"
                    >
                      {memoText ? '편집' : '메모 작성'}
                    </button>
                  )}
                </div>
                
                {isEditingMemo ? (
                  <div className="flex flex-col gap-3 animate-in fade-in duration-200">
                    <textarea
                      className="w-full p-4 rounded-xl border border-blue-300 focus:ring-4 focus:ring-blue-100 focus:border-blue-500 outline-none resize-y min-h-[120px] text-gray-700 bg-white shadow-sm transition-all"
                      placeholder="이 자료에 대한 개인적인 생각이나 인사이트를 남겨보세요..."
                      value={memoText}
                      onChange={(e) => setMemoText(e.target.value)}
                      autoFocus
                    />
                    <div className="flex justify-end gap-2">
                      <button 
                        onClick={() => { setMemoText(card.memo || ''); setIsEditingMemo(false); }} 
                        className="px-4 py-2 text-sm text-gray-600 hover:bg-gray-100 rounded-lg transition-colors font-medium"
                      >
                        취소
                      </button>
                      <button 
                        onClick={handleSaveMemo} 
                        className="px-5 py-2 text-sm bg-blue-600 text-white font-medium rounded-lg hover:bg-blue-700 transition-colors shadow-sm"
                      >
                        저장
                      </button>
                    </div>
                  </div>
                ) : (
                  <div
                    onClick={() => setIsEditingMemo(true)}
                    className={`p-5 rounded-xl border transition-all cursor-text min-h-[80px] shadow-sm
                      ${memoText ? 'bg-yellow-50/50 border-yellow-200 text-gray-800 hover:bg-yellow-50' : 'bg-gray-50 border-gray-200 text-gray-400 hover:bg-gray-100 border-dashed'}`}
                  >
                    {memoText ? (
                      <div className="whitespace-pre-wrap leading-relaxed">{memoText}</div>
                    ) : (
                      <span className="flex items-center justify-center h-full h-[40px]">
                        클릭하여 새로운 메모를 추가하세요...
                      </span>
                    )}
                  </div>
                )}
              </section>

            </div>
          </div>
          
          <div className="flex-none bg-gray-50 border-t border-gray-100 px-6 py-4 flex justify-end">
             <a 
               href={card.url} target="_blank" rel="noreferrer"
               className="flex items-center px-6 py-2.5 bg-blue-600 text-white font-medium rounded-lg hover:bg-blue-700 transition-colors shadow-sm"
             >
                원본 사이트 열기 <ExternalLink className="w-4 h-4 ml-2" />
             </a>
          </div>

        </div>
      </div>
    );
  };

  // 3. 정보 카드 컴포넌트
  const CardInfo = ({ card }) => (
    <div 
      onClick={() => setSelectedCard(card)}
      className="bg-white rounded-2xl border border-gray-200 p-6 hover:shadow-xl hover:-translate-y-1 hover:border-blue-300 transition-all cursor-pointer group flex flex-col h-[280px] relative animate-in fade-in zoom-in-95 duration-300"
    >
      <div className="absolute top-4 right-4 opacity-0 group-hover:opacity-100 transition-opacity">
        <button className="p-1.5 text-gray-400 hover:text-gray-800 hover:bg-gray-100 rounded-md" onClick={(e) => { e.stopPropagation(); }}>
           <MoreVertical className="w-5 h-5" />
        </button>
      </div>

      <div className="mb-4 flex flex-wrap gap-2 pr-8">
        {card.tags.slice(0, 3).map(tag => (
          <span key={tag} className="px-2.5 py-1 bg-blue-50/50 text-blue-700 text-xs rounded-md font-semibold border border-blue-100/50">
            {tag}
          </span>
        ))}
        {card.tags.length > 3 && <span className="px-2 py-1 text-gray-400 text-xs font-medium">+{card.tags.length - 3}</span>}
      </div>
      
      <h3 className="text-lg font-bold text-gray-900 mb-3 group-hover:text-blue-600 transition-colors line-clamp-2 leading-snug">
        {card.title}
      </h3>
      
      <p className="text-sm text-gray-500 line-clamp-3 mb-6 flex-1 leading-relaxed">
        {card.summary}
      </p>
      
      {/* 메모가 존재하는 카드면 UI에 작게 힌트 표시 */}
      {card.memo && (
        <div className="mb-4 text-xs font-medium text-yellow-600 bg-yellow-50 px-2 py-1 rounded w-fit flex items-center">
          <PenLine className="w-3 h-3 mr-1" /> 메모 있음
        </div>
      )}
      
      <div className="mt-auto pt-4 border-t border-gray-100 flex items-center justify-between text-xs">
        <div className="flex items-center text-gray-400 truncate pr-4">
          <Link className="w-3.5 h-3.5 mr-1.5 shrink-0" />
          <span className="truncate">{card.url.replace(/^https?:\/\//, '')}</span>
        </div>
        <button 
          onClick={(e) => handleCopyLink(card.url, e)}
          className="p-1.5 text-gray-400 hover:text-blue-600 hover:bg-blue-50 rounded-md transition-colors shrink-0 opacity-0 group-hover:opacity-100"
          title="링크 복사"
        >
          <Copy className="w-4 h-4" />
        </button>
      </div>
    </div>
  );

  return (
    <div className="flex flex-col h-screen bg-[#F8FAFC] font-sans text-gray-900 overflow-hidden">
      
      {/* ==========================================
          [AI-SECTION: TOP NAVIGATION]
          ========================================== */}
      <header className="h-16 bg-white border-b border-gray-200 px-4 lg:px-6 flex items-center justify-between z-20 shrink-0">
        
        <div className="flex items-center shrink-0">
          <div className="w-8 h-8 bg-blue-600 rounded-lg flex items-center justify-center mr-3 shadow-sm shadow-blue-200">
            <Bookmark className="w-5 h-5 text-white" />
          </div>
          <div className="flex flex-col justify-center">
            <span className="font-extrabold text-xl tracking-tight text-gray-900 leading-none mb-1 mt-1">
              Knowledge<span className="text-blue-600">Hub</span>
            </span>
            <div className="flex items-center gap-1 bg-green-50/80 px-1.5 py-0.5 rounded border border-green-100/50 w-fit mt-0.5">
              <span className="relative flex h-1.5 w-1.5 mr-0.5">
                <span className="animate-ping absolute inline-flex h-full w-full rounded-full bg-green-400 opacity-75"></span>
                <span className="relative inline-flex rounded-full h-1.5 w-1.5 bg-green-500"></span>
              </span>
              <Bot className="w-2.5 h-2.5 text-green-600" />
              <span className="text-[9px] font-bold text-green-700 tracking-wider uppercase">GPT Connected</span>
            </div>
          </div>
        </div>

        <div className="flex-1 max-w-3xl flex items-center gap-4 px-4 ml-6">
          <div className="relative flex-1 group">
            <div className="absolute inset-y-0 left-0 pl-4 flex items-center pointer-events-none">
              <Plus className="h-5 w-5 text-gray-400 group-focus-within:text-blue-500 transition-colors" />
            </div>
            <input
              type="text"
              placeholder="새로운 URL을 붙여넣어 지식을 저장하세요... (Enter 키 입력)"
              className="w-full pl-11 pr-24 py-2.5 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-blue-500 focus:border-blue-500 focus:bg-white transition-all text-sm"
              value={urlInput}
              onChange={(e) => setUrlInput(e.target.value)}
              onKeyDown={handleKeyDown}
              disabled={isAnalyzing}
            />
            <button 
              onClick={handleAnalyzeUrl}
              disabled={isAnalyzing}
              className={`absolute inset-y-1.5 right-1.5 px-4 rounded-lg text-sm font-semibold transition-colors shadow-sm flex items-center
                ${isAnalyzing ? 'bg-gray-400 text-white cursor-not-allowed' : 'bg-gray-900 text-white hover:bg-blue-600'}`}
            >
              {isAnalyzing ? <Loader2 className="w-4 h-4 animate-spin" /> : 'AI 분석'}
            </button>
          </div>

          <div className="relative w-72">
            <div className="absolute inset-y-0 left-0 pl-3.5 flex items-center pointer-events-none">
              <Search className="h-4 w-4 text-gray-400" />
            </div>
            <input
              type="text"
              placeholder="저장된 지식 검색..."
              className="w-full pl-10 pr-4 py-2 bg-gray-100 border-transparent border rounded-xl focus:ring-2 focus:ring-gray-200 focus:bg-white focus:border-gray-200 transition-all text-sm"
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
            />
            {searchQuery && (
              <button onClick={() => setSearchQuery('')} className="absolute inset-y-0 right-0 pr-3 flex items-center text-gray-400 hover:text-gray-600">
                <X className="w-4 h-4" />
              </button>
            )}
          </div>
        </div>

        <div className="flex items-center gap-3 shrink-0 ml-4">
          <button className="p-2 text-gray-400 hover:text-gray-700 hover:bg-gray-100 rounded-full transition-colors relative">
            <Bell className="w-5 h-5" />
          </button>
          <button className="p-2 text-gray-400 hover:text-gray-700 hover:bg-gray-100 rounded-full transition-colors">
            <Settings className="w-5 h-5" />
          </button>
          <div className="w-px h-6 bg-gray-200 mx-1"></div>
          <button className="flex items-center gap-2 pl-2 pr-1 py-1 hover:bg-gray-100 rounded-full transition-colors">
            <div className="w-8 h-8 bg-gray-800 rounded-full flex items-center justify-center text-white shadow-sm">
              <User className="w-4 h-4" />
            </div>
          </button>
        </div>
      </header>
      
      {/* ==========================================
          [AI-SECTION: MAIN CONTENT AREA]
          ========================================== */}
      <div className="flex flex-1 overflow-hidden">
        
        <aside className="w-64 bg-white border-r border-gray-200 flex flex-col z-10">
          <div className="p-4 flex-1 overflow-y-auto space-y-8">
            <div>
              <div className="flex items-center justify-between px-2 mb-3">
                <h3 className="text-xs font-bold text-gray-400 uppercase tracking-wider">내 폴더</h3>
                <button className="p-1 text-gray-400 hover:text-blue-600 hover:bg-blue-50 rounded transition-colors" title="폴더 추가">
                  <Plus className="w-4 h-4" />
                </button>
              </div>
              <ul className="space-y-1">
                {MOCK_FOLDERS.map((folder) => (
                  <li key={folder.id}>
                    <button 
                      onClick={() => setActiveFolder(folder.id)}
                      className={`w-full flex items-center px-3 py-2.5 rounded-xl text-sm font-medium transition-all ${
                        activeFolder === folder.id 
                          ? 'bg-blue-50 text-blue-700' 
                          : 'text-gray-600 hover:bg-gray-50 hover:text-gray-900'
                      }`}
                    >
                      <folder.icon className={`w-4 h-4 mr-3 ${activeFolder === folder.id ? 'text-blue-600' : 'text-gray-400'}`} />
                      {folder.name}
                      {activeFolder === folder.id && (
                        <span className="ml-auto text-xs bg-blue-100 text-blue-700 px-2 py-0.5 rounded-full">
                          {filteredCards.length}
                        </span>
                      )}
                    </button>
                  </li>
                ))}
              </ul>
            </div>
          </div>
        </aside>

        <main className="flex-1 overflow-y-auto p-6 md:p-8 relative">
          <div className="max-w-7xl mx-auto h-full flex flex-col">
            
            <div className="flex flex-wrap items-end justify-between mb-6 pb-4 border-b border-gray-200 gap-4">
              <div>
                <h1 className="text-2xl font-extrabold text-gray-900 mb-1">
                  {MOCK_FOLDERS.find(f => f.id === activeFolder)?.name}
                </h1>
                <p className="text-sm text-gray-500">
                  총 {filteredCards.length}개의 지식이 저장되어 있습니다.
                </p>
              </div>
              
              <div className="flex items-center gap-3">
                <button className="flex items-center px-3 py-2 bg-white border border-gray-200 rounded-lg text-sm font-medium text-gray-600 hover:bg-gray-50 transition-colors shadow-sm">
                  <Filter className="w-4 h-4 mr-2 text-gray-400" /> 최신순
                </button>
                <div className="flex items-center p-1 bg-gray-100 rounded-lg border border-gray-200">
                  <button className="p-1.5 bg-white text-gray-800 rounded shadow-sm">
                    <Grid className="w-4 h-4" />
                  </button>
                  <button className="p-1.5 text-gray-400 hover:text-gray-800 transition-colors">
                    <List className="w-4 h-4" />
                  </button>
                </div>
              </div>
            </div>

            {filteredCards.length > 0 || isAnalyzing ? (
              <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6 pb-12">
                {isAnalyzing && <SkeletonCard />}
                
                {filteredCards.map(card => (
                  <CardInfo key={card.id} card={card} />
                ))}
              </div>
            ) : (
              <div className="flex-1 flex flex-col items-center justify-center text-center px-4 py-20 opacity-80">
                <div className="w-20 h-20 bg-gray-100 rounded-full flex items-center justify-center mb-4">
                  <Search className="w-10 h-10 text-gray-400" />
                </div>
                <h3 className="text-lg font-bold text-gray-900 mb-2">결과가 없습니다</h3>
                <p className="text-gray-500 max-w-sm">
                  {searchQuery ? `'${searchQuery}'에 해당하는 지식을 찾을 수 없습니다.` : '현재 폴더에 저장된 지식이 없습니다.'}
                </p>
              </div>
            )}
          </div>
        </main>
      </div>

      {selectedCard && (
        <DetailModal 
          card={selectedCard} 
          onClose={() => setSelectedCard(null)} 
          onUpdateCard={handleUpdateCard} 
        />
      )}
    </div>
  );
}
